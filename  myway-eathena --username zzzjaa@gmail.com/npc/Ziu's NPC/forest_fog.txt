	areamonster "1@mist",254,123,259,118,"Parus",2134,2,instance_npcname("#mobfogins", instance_id())+"::OnMyMobDead";
	}

	areamonster "1@mist",254,123,259,118,"Little Fatum",2136,2,instance_npcname("#mobfogins", instance_id())+"::OnMyMobDead";
	}
	if (.@mob_ran > 20) {
	areamonster "1@mist",254,123,259,118,"Miming",2137,2,instance_npcname("#mobfogins", instance_id())+"::OnMyMobDead";
	}

	set .@callnum,2;
	set .@e_mob_num,mobcount("1@mist",instance_npcname("#mobfogins", instance_id())+"::OnMyMobDead");
